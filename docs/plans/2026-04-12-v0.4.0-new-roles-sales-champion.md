# src/services/calculator.py 新增方法

def calculate_all(self) -> Dict[str, CommissionResult]:
    """计算所有人提成，包含销冠判定"""
    results = {}
    
    # 先计算所有人基础提成
    for person in self.people.values():
        results[person.id] = self.calculate_person(person)
    
    # 判定销冠
    self._assign_sales_champion_bonus(results)
    
    # 更新总提成
    for result in results.values():
        result.total = result.personal_commission + result.team_commission + result.management_bonus + result.high_performance_bonus + result.sales_champion_bonus
    
    return results

def _assign_sales_champion_bonus(self, results: Dict[str, CommissionResult]):
    """判定销冠并分配奖金"""
    threshold = self.config.sales_champion_threshold
    bonus = self.config.sales_champion_bonus
    
    # 找出最高业绩者
    max_performance = 0.0
    champion_id = None
    
    for person in self.people.values():
        if person.performance >= threshold and person.performance > max_performance:
            max_performance = person.performance
            champion_id = person.id
    
    # 分配销冠奖金
    if champion_id:
        results[champion_id].sales_champion_bonus = bonus
        logger.info(f"销冠: {self.people[champion_id].name}, 业绩={max_performance}, 奖金={bonus}")
```

def calculate_person(self, person: Person) -> CommissionResult:
    result = CommissionResult(person_id=person.id)
    
    # 个人提成
    result.personal_commission = calculate_personal_commission(...)
    
    # 记录提成点数（比例×100，4位小数）
    rate = self._get_personal_commission_rate(person.performance)
    result.commission_rate = round(rate * 100, 4)  # 如20%→20.0000 或 0.2000
    
    ...
    return result

def _get_personal_commission_rate(self, performance: float) -> float:
    """获取个人提成比例"""
    for tier in self.config.personal_commission.tiers:
        if tier.min_amount <= performance:
            if tier.max_amount is None or performance < tier.max_amount:
                return tier.rate
    return 0.0
```
